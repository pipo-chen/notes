209. 长度最小的子数组

209. 长度最小的子数组

 public int minSubArrayLen(int s, int[] nums) {
        int len = nums.length + 1;
        int left = 0, right = 0, sum = 0;
        while (right < nums.length) {
            sum += nums[right++];
            while (sum >= s) {
                len = Math.min(len, right - left);
                sum -= nums[left++];
            }
        }
        return len == nums.length + 1 ? 0 : len;
    }
